CREATE TABLE Alimento (
    Id INT PRIMARY KEY,
    Nome VARCHAR(100) NOT NULL,
    Quantidade FLOAT,
    Energia FLOAT,
    Gordura FLOAT,
    Carboidratos FLOAT,
    Proteina FLOAT,
    Colesterol FLOAT,
    FibraAlimentare FLOAT
);

CREATE TABLE GruposAlimento (
    Id INT PRIMARY KEY,
    IdAlimento INT,
    Descricao VARCHAR(255) NOT NULL,
    FOREIGN KEY (IdAlimento) REFERENCES Alimento(Id)
);
